# Зависимости проекта. Все ставятся колёсами (wheels) из коробки — без GDAL/компиляции,
# чтобы судья поднял окружение с первого `pip install -r requirements.txt`.

# --- Ядро (компьютерное зрение) ---
opencv-python      # чтение картинок, HSV, морфология, контуры
numpy              # массивы/матрицы — на них опирается OpenCV

# --- Геопривязка (Трек 2: пиксели -> WGS84) ---
pyproj             # трансформация датума Пулково 1942 (EPSG:4284) -> WGS84 (EPSG:4326)

# --- Векторизация (осевые линии вместо контуров-петель) ---
scikit-image       # скелетизация маски в линию 1px; при отсутствии — фолбэк на контуры

# --- Опционально: экспорт Shapefile (бонус-формат к GeoJSON) ---
pyshp              # запись .shp/.dbf/.shx; если не установлен — пишется только GeoJSON

# --- Тесты ---
pytest             # быстрый smoke-тест (python -m pytest)
