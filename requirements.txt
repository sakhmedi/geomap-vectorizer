# Зависимости проекта. Все ставятся колёсами (wheels) из коробки — без GDAL/компиляции,
# чтобы судья поднял окружение с первого `pip install -r requirements.txt`.

# Версии закреплены по мажору (на чём проверено), чтобы у судьи на свежем окружении
# не подтянулся ломающий релиз. Диапазоны, а не точные пины — чтобы колесо нашлось
# под разные версии Python (проверено на 3.9–3.14).

# --- Ядро (компьютерное зрение) ---
opencv-python>=4.8,<5      # чтение картинок, HSV, морфология, контуры
numpy>=1.24,<3             # массивы/матрицы — на них опирается OpenCV

# --- Геопривязка (Трек 2: пиксели -> WGS84) ---
pyproj>=3.5,<4             # трансформация датума Пулково 1942 (EPSG:4284) -> WGS84 (EPSG:4326)

# --- Векторизация (осевые линии вместо контуров-петель) ---
scikit-image>=0.21,<1.0    # скелетизация маски в линию 1px; при отсутствии — фолбэк на контуры

# --- Опционально: экспорт Shapefile (бонус-формат к GeoJSON) ---
pyshp>=2.3,<3              # запись .shp/.dbf/.shx; если не установлен — пишется только GeoJSON

# --- Тесты ---
pytest>=7.0               # быстрый smoke-тест (python -m pytest)
